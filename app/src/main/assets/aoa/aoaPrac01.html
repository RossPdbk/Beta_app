#include<stdio.h>
#include<conio.h>
#define MAX 7
void knapsack(float x[],float w[],float p[],int n,int m,int in[])
{
	int i;
	float weight,profit;
	weight=0.0;
	profit=0.0;
	for(i=0;i<n&&weight<m;i++)
	{
		if(weight+w[in[i]]<m)
		{
			x[in[i]]=1;
			weight+=w[in[i]];
		}
		else
		{
			x[in[i]]=((m-weight)/w[in[i]]);
			weight+=x[in[i]]*w[in[i]];
		}
		profit+=x[in[i]]*p[in[i]];
	}
}
void main()
{
	int n,i,j,m,t,in[MAX]={0};
	float ratio[MAX]={0.00},x[MAX]={0.00},p[MAX],w[MAX],temp;
	clrscr();
	printf("\nEnter no. of objects:");
	scanf("%d",&n);
	printf("\nEnter weight & profit of each object.");;
	for(i=0;i<n;i++)
	{
		printf("\nobject-%d\nweight[%d]=",i+1,i+1);
		scanf("%f",&w[i]);
		printf("Profit[%d]=",i+1);
		scanf("%f",&p[i]);
    }
	printf("\nEnter capacity of knapsack:");
	scanf("%d",&m);
	for(i=0;i<n;i++)
	{
		ratio[i]=p[i]/w[i];
		in[i]=i;
    }
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(ratio[i]<ratio[j])
			{
				temp=ratio[i];
				ratio[i]=ratio[j];
				ratio[j]=temp;
				t=in[i];
				in[i]=in[j];
				in[j]=t;
			}
		}
    }
	knapsack(x,w,p,n,m,in);
	printf("\nFraction of objects for maximum profit :(");
	for(i=0;i<n;i++)
    {
		printf("%.2f,",x[i]);
    }
	printf(")");
	getch();
}
