Insertion Sort 
#include <stdio.h>
void insertiosort(int array[],int n)
{
	int c,d,t;
	for(c = 1 ; c <= n - 1; c++)
	{
		d = c;
		while ( d > 0 && array[d] < array[d-1])
		{
			t = array[d];
			array[d]   = array[d-1];
			array[d-1] = t;
			d--;
		}
	}
}

void main()
{
	int n, array[1000], c, d, t;
	printf("Enter number of elements\n");
	scanf("%d", &n);
	printf("Enter %d integers\n", n);
	for (c = 0; c < n; c++)
	{
		scanf("%d", &array[c]);
	}	
	insertiosort(array,n);
	printf("Sorted list :\n");
	for (c = 0; c <= n - 1; c++)
	{
		printf("%d\n", array[c]);
	} 
	getch();
}
