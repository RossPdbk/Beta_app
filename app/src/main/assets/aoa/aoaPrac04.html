Selection Sort

#include<stdio.h>
#include<conio.h>
void main()
{
    int i,a[100],n;
    clrscr();
    printf("Enter the number of elements :");
    scanf("%d",&n);
    printf("\nEnter the elements :");
    for(i=0;i<n;i++)
    {
        printf("\nElement %d :",i+1);
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            if(a[j]<a[i])
            {
                int temp=a[j];
                a[j]=a[i];
                a[i]=temp;
            }
        }
    }
    printf("Sorted elements :");
    for(i=0;i<n;i++)
        printf(" %d",a[i]);
    getch();
}
