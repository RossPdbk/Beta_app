#include<stdio.h>
#include<conio.h>
int partition(int [],int,int);
void quicksort(int [],int,int);
void main()
{
	int a[20],i,n;
	printf("Length :");
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		printf("elemenr %d :",i+1);
		scanf("%d",&a[i]);
	}
	quicksort(a,0,n-1);
	printf("After sorting :");
	for(i=0;i<n;i++)
		printf("%d ",a[i]);
	getch();
}
void quicksort(int a[],int start,int end )
{
	if(start<end)
	{
		int j;
		j=partition(a,start,end);
		quicksort(a,start,j-1);
		quicksort(a,j+1,end);
	}
}
int partition(int a[],int start,int end)
{
	int j,pivot,i,temp;
	i=start;
	j=end;
	pivot=start;
	while(i<j)
	{
		while(a[i]<=a[pivot] && i<end)
			i++;
		while(a[j]>a[pivot])
			j--;
		if(i<j)
		{
		temp=a[j];
		a[j]=a[i];
		a[i]=temp;
		}
	}
	temp=a[pivot];
	a[pivot]=a[j];
	a[j]=temp;
	return j;
}
