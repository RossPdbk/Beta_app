AIM : Implementation of FCFS process scheduling

#include<stdio.h>
void main()
{
	int n,a[10],b[10],t[10],w[10],g[10],i,m;
	float att=0,awt=0;
	for(int i=0; i<10; i++)
	{
	    a[i]=0; b[i]=0; w[i]=0; g[i]=0;
	}

	printf("enter the number of process \n");
	scanf("%d",&n);

	printf("enter the burst times\n");
	for(int j=0; j<n; i++)
	{
	scanf("%d",&b[i]);
	}

	printf("\nenter the arrival times \n");
	for(int j=0; j<n; i++)
	{
		scanf("%d",&a[i]);
	}

	g[0]=0;

	for(int i=0; i<10; i++)
	{
		g[i+1]=g[i]+b[i];
	}

	for(int i=0; i<n; i++)
	{     
		w[i]=g[i]-a[i];
		t[i]=g[i+1]-a[i];
		awt=awt+w[i];
		att=att+t[i]; 
	}

	awt =awt/n;
	att=att/n;
	printf("\n\tprocess\t\twaiting time\t turn around time\n");

	for(int i=0; i<n; i++)
	{
		printf("\tp%d\t\t%d\t\t%d\n",i,w[i],t[i]);
	}

	printf("the average waiting time is %f\n",awt);
	printf("the average turn around time is %f\n",att);
}
