AIM : Implementation of Best fit algorithm in memory management.

 
public class BestFit 
{
	static void bestFit(int partitionSize[], int m, int processSize[], int n)
	{
		int allocation[] = new int[n];

		for (int i = 0; i < allocation.length; i++)
			allocation[i] = -1;
      
		for (int i=0; i<n; i++)
		{
			int bestIndex = -1;
			for (int j=0; j<m; j++)
			{
				if (partitionSize[j] >= processSize[i])
				{
					if (bestIndex == -1)
						bestIndex = j;
					else if (partitionSize[bestIndex] > partitionSize[j])
						bestIndex = j;
				}
			}

			if (bestIndex != -1)
			{
				allocation[i] = bestIndex;
				partitionSize[bestIndex] -= processSize[i];
			}
		}
      
		System.out.println("\nProcess No.\tProcess Size\tBlock no.");
		for (int i = 0; i < n; i++)
		{
			System.out.print("   " + (i+1) + "\t\t\t" + processSize[i] + "\t\t\t\t");
			if (allocation[i] != -1)
				System.out.print(allocation[i] + 1);
			else
				System.out.print("Not Allocated");
			System.out.println();
		}
	}
	
	public static void main(String[] args)
	{
		int partitionSize[] = {100, 500, 200, 300, 600};
		int processSize[] = {212, 417, 112, 426};
		int m = partitionSize.length;
		int n = processSize.length;
          
		bestFit(partitionSize, m, processSize, n);
	}
}
