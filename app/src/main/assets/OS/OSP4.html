AIM : Implementation of FCFS disk scheduling algorithm.

import java.util.Scanner;
class FCFS{
	public static void main(String args[]){
		Scanner sc=new Scanner(System.in);
		int nd,tft=0,hs;

		System.out.println("Enter no. of disks:");
		nd=sc.nextInt();
		int ft[]=new int[nd+1];

		System.out.println("\nEnter starting position of Head:");
		hs=sc.nextInt();
		ft[0]=hs;

		System.out.println("\nEnter disk tracks:");
		for(int i=0;i<nd;i++)
		{
			ft[i+1]=sc.nextInt();
		}

	System.out.print("\nHead move Sequence: [ ");
	for(int i=0;i<nd;i++)
	{
		System.out.print(ft[i]+" -> ");
		if(ft[i]<ft[i+1])
		{
			tft+=ft[i+1]-ft[i];
		}
		else
		{
			tft+=ft[i]-ft[i+1];
		}
	}
	System.out.print(ft[nd]+" ]");
	System.out.println("\nTotal sleek time = "+tft);
	}
}
