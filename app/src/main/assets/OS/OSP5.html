AIM : Implementation of SSTF disk scheduling algorithm.

import java.util.Scanner;
class SSTF{
	public static void main(String args[]){
		Scanner sc=new Scanner(System.in);
		int nd,tft=0,hs,hp=0,np,pp,nhp;

		System.out.println("Enter no. of disks:");
		nd=sc.nextInt();
		int ft[]=new int[nd+1];

		System.out.println("Enter starting position of Head:");
		hs=sc.nextInt();

		System.out.println("Enter disk tracks:");
		for(int i=0;i<nd;i++)
		{
			ft[i]=sc.nextInt();
		}	
		
		ft[nd]=hs;	

		for(int i=0;i<nd;i++)
		{
			for(int j=0;j<nd;j++)
			{
				if(ft[j]>ft[j+1])
				{
					int temp=ft[j];
					ft[j]=ft[j+1];
					 ft[j+1]=temp;
				}
			}
		}	

		for(int i=0;i<=nd;i++)
		{
			if(ft[i]==hs)
			{
				hp=i;
				break;
			}
		}

		np=hp+1;
		pp=hp-1;

		System.out.print("Head move Sequence: [ ");
		for(int i=nd;i>0;i--)
		{
			System.out.print(ft[hp]+" -> ");
			if(np==nd+1)
			{
				tft+=ft[hp]-ft[pp];
				hp=pp;
				pp--;
			}
			else if(pp==-1)
			{
				tft+=ft[np]-ft[hp];
				hp=np;
				np++;
			}
			else if((ft[hp]-ft[pp])<(ft[np]-ft[hp]))
			{
				tft+=ft[hp]-ft[pp];
				pp--;
			}
			else if(np<=nd)
			{
				tft+=ft[np]-ft[hp];
				hp=np;
				np++;
			}
		}	
		System.out.print(ft[nd]+" ]");
		System.out.println("\nTotal sleek time = "+tft);
	}
}
